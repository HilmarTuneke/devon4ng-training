Functions

			
function add(x, y) {
	var total = x + y;
	return total;
}

add(); // NaN
// You can't perform addition on undefined

add(2, 3, 4); // 5
// added the first two; 4 was ignored

visibility is an advanced topic

Functions can be stored in variables

			
var isPositive = function(value) {
	return value > 0;
}
			
		
			
isPositive(); // calls the function
isPositive; // points to the function itself

parantheses: call the function, no parantheses: pass the function

Functions can be passed

			
myArray.filter(function(current){ // anonymous function
	return current > 0; // remove all negative values from array
});

// same but reusable function
var isPositive = function(value) {
	return value > 0;
}
myArray.filter(isPositive); // remember: don't use 'myArray.filter(isPositive());' !!
			
		
			
// function as return value
angular.module('app.table-mgmt').factory('tables', function (tableManagementRestService) {
	return {
		getPaginatedTables: function (pagenumber, pagesize) { ... },
		loadTable: function (tableId) { ... },
		cancelReservation: function (table) { ... },
	}
});

Objects

no classes, no interfaces


var obj = {};
		

var person = {
	name = 'John'
};
person.lastname = 'Doe'; // modify object after declaration
		

var obj = {
  name: "Carrot",
  "for": "Max",
  details: {
    color: "orange",
    size: 12
  }
}

// access properties
obj.details.color; // orange
obj["details"]["size"]; // 12

Object + Functions

			
var batman = {
	firstname = 'Bruce',
	lastname = 'Wayne',
	fullname = function() {
		return this.firstname + ' ' + this.lastname;
	}
}

Be careful with 'this' though, it depends on the scope (advanced topic).
