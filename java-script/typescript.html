let

Solves most hoisting / scope issues by defining a block scope

					
let foo = true;
if  (foo) {
    let bar = 'baz';
    console.log(bar); // outputs 'baz'
}

try {
    console.log(bar);
} catch (e) {
    console.log("bar doesn't exist");
}
				
			
					
for (let i = 0; i <= 2; i += 1) {
    console.log(i); // current i
}
try {
    console.log(i);
} catch (e) {
    console.log("i doesn't exist");
}
